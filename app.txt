import React, { useState, useEffect } from 'react';
import { View, Text, FlatList, StyleSheet, RefreshControl, TouchableOpacity, ScrollView, ActivityIndicator, Modal, PermissionsAndroid, Platform, Alert, Dimensions } from 'react-native';
import { SafeAreaView, SafeAreaProvider } from 'react-native-safe-area-context';
import YoutubePlayer from "react-native-youtube-iframe";
import axios from 'axios';
import AsyncStorage from '@react-native-async-storage/async-storage';
import messaging from '@react-native-firebase/messaging';
const screenHeight = Dimensions.get('window').height;

const AVAILABLE_LANGUAGES = [
  { id: 'hindi', label: 'Hindi' },
  { id: 'english', label: 'English' },
  { id: 'chhattisgarhi', label: 'Chhattisgarhi' },
  { id: 'marathi', label: 'Marathi' },
  { id: 'telugu', label: 'Telugu' },
  { id: 'punjabi', label: 'Punjabi' }
];

const App = () => {
  const [videos, setVideos] = useState([]);
  const [loading, setLoading] = useState(true);
  
  const [activeTab, setActiveTab] = useState('video'); 
  const [activeCategory, setActiveCategory] = useState('all'); 
  const [activeLanguage, setActiveLanguage] = useState('my_languages'); 
  const [showLangDropdown, setShowLangDropdown] = useState(false);

  const [appReady, setAppReady] = useState(false); 
  const [hasOnboarded, setHasOnboarded] = useState(false);
  const [userLanguages, setUserLanguages] = useState([]); 
  const [tempLanguages, setTempLanguages] = useState([]);
  const [fullScreenVideoId, setFullScreenVideoId] = useState(null);

  const requestUserPermission = async () => {
    if (Platform.OS === 'android' && Platform.Version >= 33) {
      await PermissionsAndroid.request(PermissionsAndroid.PERMISSIONS.POST_NOTIFICATIONS);
    }
    const authStatus = await messaging().requestPermission();
    const enabled = authStatus === messaging.AuthorizationStatus.AUTHORIZED || authStatus === messaging.AuthorizationStatus.PROVISIONAL;
    if (enabled) {
      const token = await messaging().getToken();
      console.log("🔥 YE RAHA AAPKA FCM TOKEN:", token);
    }
    messaging().onMessage(async remoteMessage => {
      Alert.alert('🔥 Naya Trend Aaya Hai!', remoteMessage.notification.body);
    });
  };

  const API_URL = 'http://192.168.29.79:5000/videos';

  useEffect(() => {
    checkUserMemory();
    fetchVideos();
    requestUserPermission();
  }, []);

  const checkUserMemory = async () => {
    try {
      const savedLangs = await AsyncStorage.getItem('@trendz_languages');
      if (savedLangs !== null) {
        setUserLanguages(JSON.parse(savedLangs));
        setHasOnboarded(true);
      }
      setAppReady(true);
    } catch (e) {
      setAppReady(true);
    }
  };

  const finishOnboarding = async () => {
    if (tempLanguages.length === 0) return; 
    try {
      await AsyncStorage.setItem('@trendz_languages', JSON.stringify(tempLanguages));
      setUserLanguages(tempLanguages);
      setHasOnboarded(true);

      tempLanguages.forEach(lang => {
        messaging().subscribeToTopic(lang)
          .then(() => console.log(`✅ FIREBASE: Hum ${lang.toUpperCase()} ke group mein add ho gaye!`));
      });
    } catch (e) {
      console.log("Save error", e);
    }
  };

  const fetchVideos = async () => {
    try {
      const response = await axios.get(API_URL);
      setVideos(response.data);
      setLoading(false);
    } catch (error) {
      setLoading(false);
    }
  };

  const calculateTimeLeft = (createdAt) => {
    const timeLeft = (new Date(createdAt).getTime() + (72 * 60 * 60 * 1000)) - new Date().getTime();
    if (timeLeft <= 0) return "Expired";
    return `${Math.floor(timeLeft / (1000 * 60 * 60))}h ${Math.floor((timeLeft % (1000 * 60 * 60)) / (1000 * 60))}m`;
  };

  const filteredVideos = videos.filter((video) => {
    const isTabMatch = video.type === activeTab;
    const isCategoryMatch = activeCategory === 'all' || video.category === activeCategory;
    
    let isLanguageMatch = false;
    if (activeLanguage === 'my_languages') {
      isLanguageMatch = userLanguages.includes(video.language) || !video.language;
    } else if (activeLanguage === 'all') {
      isLanguageMatch = true;
    } else {
      isLanguageMatch = video.language === activeLanguage;
    }
    
    return isTabMatch && isCategoryMatch && isLanguageMatch;
  });

  const renderVideoItem = ({ item }) => {
    const isShort = item.type === 'short';
    const videoHeight = isShort ? 320 : 220; 

    return (
      <View style={styles.card}>
        {/* Direct YouTube Player: Ek click aur video chalu! */}
        <View style={{ height: videoHeight, width: '100%', backgroundColor: '#000' }}>
           <YoutubePlayer 
             height={videoHeight} 
             play={false} 
             videoId={item.videoUrl} 
           />
        </View>
        
        {/* Sleek Text Container */}
        <View style={styles.textContainer}>
          <Text style={styles.title} numberOfLines={2}>{item.title}</Text>
          
          <View style={styles.detailsRow}>
            <Text style={styles.channel}>{item.channelName}</Text>
            <Text style={styles.timer}>⏳ Expires in {calculateTimeLeft(item.createdAt)}</Text>
          </View>
          
          {item.language && <Text style={styles.langTag}>{item.language.toUpperCase()}</Text>}
        </View>
      </View>
    );
  };

  if (!appReady || loading) {
    return (
      <View style={{ flex: 1, backgroundColor: '#121212', justifyContent: 'center', alignItems: 'center' }}>
        <ActivityIndicator size="large" color="#ff4444" />
      </View>
    );
  }

  if (!hasOnboarded) {
    return (
      <SafeAreaView style={styles.onboardContainer}>
        <Text style={styles.onboardTitle}>Kya dekhna pasand karenge? 🍿</Text>
        <Text style={styles.onboardSub}>Apni pasandida bhashayein chunein:</Text>
        <View style={styles.grid}>
          {AVAILABLE_LANGUAGES.map((lang) => {
            const isSelected = tempLanguages.includes(lang.id);
            return (
              <TouchableOpacity key={lang.id} style={[styles.langBox, isSelected && styles.langBoxSelected]}
                onPress={() => {
                  if (isSelected) setTempLanguages(tempLanguages.filter(l => l !== lang.id));
                  else setTempLanguages([...tempLanguages, lang.id]);
                }}>
                <Text style={[styles.langText, isSelected && styles.langTextSelected]}>{lang.label}</Text>
              </TouchableOpacity>
            )
          })}
        </View>
        <TouchableOpacity style={[styles.continueBtn, tempLanguages.length === 0 && {opacity: 0.5}]} 
          onPress={finishOnboarding} disabled={tempLanguages.length === 0}>
          <Text style={styles.continueText}>Start Watching 🔥</Text>
        </TouchableOpacity>
      </SafeAreaView>
    );
  }

  const getActiveLanguageLabel = () => {
    if (activeLanguage === 'my_languages') return '🌟 For You';
    if (activeLanguage === 'all') return '🌍 All';
    const found = AVAILABLE_LANGUAGES.find(l => l.id === activeLanguage);
    return found ? found.label : 'Language';
  };

  return (
    <SafeAreaProvider>
      <SafeAreaView style={styles.container}>
        
        {/* Header */}
        <View style={styles.header}>
          <Text style={styles.headerTitle}>🔥 Trendz'72</Text>
          <TouchableOpacity style={styles.headerLangBtn} onPress={() => setShowLangDropdown(true)}>
            <Text style={styles.headerLangText}>{getActiveLanguageLabel()} ▾</Text>
          </TouchableOpacity>
        </View>

        {/* 🔘 ROW 1: Videos / Shorts (Size Chhota Kiya Hai) */}
        <View style={styles.tabContainer}>
          <TouchableOpacity style={[styles.tabButton, activeTab === 'video' && styles.activeTab]} onPress={() => setActiveTab('video')}>
            <Text style={[styles.tabText, activeTab === 'video' && styles.activeTabText]}>📺 Videos</Text>
          </TouchableOpacity>
          <TouchableOpacity style={[styles.tabButton, activeTab === 'short' && styles.activeTab]} onPress={() => setActiveTab('short')}>
            <Text style={[styles.tabText, activeTab === 'short' && styles.activeTabText]}>📱 Shorts</Text>
          </TouchableOpacity>
        </View>

        {/* 🏷️ ROW 2: Categories */}
        <View style={styles.categoryContainer}>
          <ScrollView horizontal showsHorizontalScrollIndicator={false}>
            {['all', 'trailer', 'music', 'drama', 'travel', 'architecture', 'mystery', 'history', 'sports'].map((cat) => (
              <TouchableOpacity key={cat} style={[styles.catButton, activeCategory === cat && styles.activeCat]} onPress={() => setActiveCategory(cat)}>
                <Text style={[styles.catText, activeCategory === cat && {color: '#000'}]}>{cat.toUpperCase()}</Text>
              </TouchableOpacity>
            ))}
          </ScrollView>
        </View>

        {/* 📺 ROW 3: Videos */}
        <FlatList
          data={filteredVideos}
          keyExtractor={(item) => item._id}
          renderItem={renderVideoItem}
          refreshControl={<RefreshControl refreshing={loading} onRefresh={fetchVideos} tintColor="#ff4444" />}
          ListEmptyComponent={<Text style={styles.emptyText}>Is category ya bhasha mein abhi kuch naya nahi hai! 😴</Text>}
        />

        {/* 🛸 THE DROPDOWN MODAL */}
        <Modal visible={showLangDropdown} transparent={true} animationType="fade">
          <TouchableOpacity style={styles.modalOverlay} activeOpacity={1} onPress={() => setShowLangDropdown(false)}>
            <View style={styles.dropdownContainer}>
              <Text style={styles.dropdownTitle}>Select Language</Text>
              <ScrollView>
                <TouchableOpacity style={[styles.dropdownItem, activeLanguage === 'my_languages' && styles.dropdownItemSelected]}
                  onPress={() => { setActiveLanguage('my_languages'); setShowLangDropdown(false); }}>
                  <Text style={[styles.dropdownItemText, activeLanguage === 'my_languages' && {color: '#ff4444', fontWeight: 'bold'}]}>🌟 For You</Text>
                </TouchableOpacity>

                <TouchableOpacity style={[styles.dropdownItem, activeLanguage === 'all' && styles.dropdownItemSelected]}
                  onPress={() => { setActiveLanguage('all'); setShowLangDropdown(false); }}>
                  <Text style={[styles.dropdownItemText, activeLanguage === 'all' && {color: '#ff4444', fontWeight: 'bold'}]}>🌍 All</Text>
                </TouchableOpacity>

                {AVAILABLE_LANGUAGES.map((lang) => (
                  <TouchableOpacity key={lang.id} style={[styles.dropdownItem, activeLanguage === lang.id && styles.dropdownItemSelected]}
                    onPress={() => { setActiveLanguage(lang.id); setShowLangDropdown(false); }}>
                    <Text style={[styles.dropdownItemText, activeLanguage === lang.id && {color: '#ff4444', fontWeight: 'bold'}]}>{lang.label}</Text>
                  </TouchableOpacity>
                ))}
              </ScrollView>
            </View>
          </TouchableOpacity>
        </Modal>

      </SafeAreaView>
    </SafeAreaProvider>
  );
};
